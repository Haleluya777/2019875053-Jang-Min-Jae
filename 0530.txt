args.c
#include <stdio.h>
#include <stdlib.h>

int main(int argc, char *argv[])
{
        int i;
        for(i = 0; i < argc; i++)
        {
                printf("argv[%d]: %s\n",i,argv[i]);
        }
        exit(0);
}


environ.c
#include <stdio.h>
#include <stdlib.h>

int main(int argc, char *argv[])
{
        char **ptr;
        extern char **environ;

        for(ptr = environ; *ptr != 0; ptr++)
        {
                printf("%s \n",*ptr);
        }
        exit(0);
}


printenv.c
#include <stdio.h>
#include <stdlib.h>
/* 환경 변수를 3개 프린트한다. */
int main(int argc, char *argv[])
{
        char *ptr;
        ptr = getenv("HOME");
        printf("HOME = %s \n", ptr);
        ptr = getenv("SHELL");
        printf("SHELL = %s \n", ptr);
        ptr = getenv("PATH");
        printf("PATH = %s \n", ptr);
        exit(0);
}
